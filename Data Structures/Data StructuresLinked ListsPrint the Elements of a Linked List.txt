/*
 * For your reference:
 *
 * SinglyLinkedListNode {
 *     int data;
 *     SinglyLinkedListNode* next;
 * };
 *
 */
void printLinkedList(SinglyLinkedListNode* head) {
SinglyLinkedListNode * temp=head;
while(temp!=NULL)
{
    cout<<temp->data<<"\n";
    temp=temp->next;
}    

}
